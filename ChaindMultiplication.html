#include <iostream>
using namespace std;
int mul[] = { 40, 20, 30, 10, 30 };
int f(int i, int j) {
    if (i == j) {
        return 0;
    }
    int answer = INT_MAX;
    int q;
    for (int k = i; k < j; k++) {
        q = f(i, k) + f(k + 1, j) + (mul[i-1] * mul[k] * mul[j]);
        answer = min(answer, q);
    }
    return answer;
}
int main() {
    cout << f(1, 4);
}
