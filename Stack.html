Here is a sample C++ code for a stack and its operations:

#include <iostream>
using namespace std;

class Stack {
private:
    int* arr;
    int current;
    int cap;
public:
    Stack(int num) {
        cap = num;
        arr = new int[num];
        current = 0;
}
    bool full() {
        if (current >= cap) {
            return true;
        } else
            return false;
}
    bool empty() {
        if (current == 0) {
            return true;
        } else
            return false;
}
    void push(int x) {
        if (!full()) {
            arr[current] = x;
            current++;
        } else {
            cout << "the stack is full ";
        }
}
    int pop() {
        if (!empty()) {
            current--;
            return arr[current];
        } else
            cout << "the stack is empty ";
        return 0;
}
};
void printStackAsFifo(Stack s, int size) {
    Stack s2(size);
    while (!s.empty()) {
        s2.push(s.pop());
}
    while (!s2.empty()) {
        cout << s2.pop() << "->";
}
}

void insertionSort(int* arr, int size) {
    int j, key;
    for (int i = 1; i < size; i++) {
        j = i - 1;
        key = arr[i];
        while (key < arr[j] && j >= 0) {
            arr[j + 1] = arr[j];
            j--;
}
        arr[j + 1] = key;
}
}
void orderStack(Stack& s, int size) {
    int* arr = new int[size];
    int i = 0;
    while (!s.empty()) {
        arr[i] = s.pop();
        i++;
}
insertionSort(arr, size);
i = 0;
while (i < size) {
        s.push(arr[i]);
        i++;
}
delete[] arr;
}

int main() {
    int size = 8;
    Stack s(size);
    s.push(10);
    s.push(11);
    s.push(8);
    s.push(12);
    s.push(54);
    s.push(39);
    s.push(2324);

orderStack(s, size);

}
