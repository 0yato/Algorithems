#include <iostream>
using namespace std;
int b;
int* arr;
int f(int des) {
    if (arr[des] == 0) {
        return 0;
    }
    if (des == b - 1) {
        return 1;
    }
    for (int k = 1; k <= arr[des]; k++) {
        if (des + k >= b) {
            break;
        }
        if (f(des + k) == 1) {
            return 1;
        }
    }
    return 0;
}
int main() {
    cin >> b;
    arr = new int[b];
    for (int i = 0; i < b; i++) {
        cin >> arr[i];
    }
    cout << f(0);
}
